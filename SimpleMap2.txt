    <script>
        alert("欢迎您！")
        var map;
        function showMap(){
            require(["esri/map",
                     "dojo/dom","dojo/on","esri/geometry/Point",
                     "dojo/domReady!"], 
              function(Map,dom,on,Point) {
                on (dom.byId("queryIt"),"click",getValues);
          map = new Map("map", {
          basemap: "topo",  //For full list of pre-defined basemaps, navigate to http://arcg.is/1JVo6Wd
          center: [-122.45,37.75], // longitude, latitude
          zoom: 13
        });
                function getValues(){
                    var mapZoom = dom.byId("mapZoom").value;
                    var mapLon = dom.byId("mapLon").value;
                    var mapLat = dom.byId("mapLat").value;
                    var mapBasemap = dom.byId("mapBasemap").value;
                    var mapPoint = new Point(mapLon,mapLat);
                    map.setZoom(mapZoom);
                    map.centerAt(mapPoint);
                    map.setBasemap(mapBasemap);
                }
      });
        }
    </script>
